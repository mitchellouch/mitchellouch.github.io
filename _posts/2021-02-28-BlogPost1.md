```javascript
$(document).ready(function () {
  //loads the event listener when the page loads
  $("#clientSearch").keyup(function () {
    //adds an event listener for keyup in the search box
    $.getJSON("res/ClientData.json", function (client) {
      //retrieves client objects from .json file
      let search = $("#clientSearch").val(); //sets search to the value in the search field
      let regex = new RegExp(search, "i"); //creates a regular expression with the value of the search field
      let clientList;

      $.each(client, function (index, clientObject) {
        //basically a for loop, parses the .json file only returns objects that contain the regex
        if (
          clientObject.lastName.search(regex) !== -1 ||
          clientObject.phone.search(regex) !== -1
        ) {
          clientList += "<tr>"; //I think I could use .append for this but I couldn't get it working
          clientList += "<td>" + clientObject.id + "</td>";
          clientList +=
            "<td>" +
            clientObject.lastName +
            ", " +
            clientObject.firstName +
            "</td>";
          clientList += "<td>" + clientObject.address + "</td>";
          clientList += "<td>" + clientObject.postalCode + "</td>";
          clientList += "<td>" + clientObject.phone + "</td>";
          clientList += "<td>" + clientObject.type + "</td>";
          clientList += "</tr>";
        }
      });
      $("tbody").html(clientList);
    });
  });
  $("#idSearch").keyup(function () {
    $.getJSON("res/ClientData.json", function (client) {
      //had to make a separate search field for id because the parameters were too overlapping
      let search = $("#idSearch").val(); //so you could never just get an id by searching id
      let clientList;

      $.each(client, function (index, clientObject) {
        if (Number(search) === Number(clientObject.id)) {
          clientList += "<tr>";
          clientList += "<td>" + clientObject.id + "</td>";
          clientList +=
            "<td>" +
            clientObject.lastName +
            ", " +
            clientObject.firstName +
            "</td>";
          clientList += "<td>" + clientObject.address + "</td>";
          clientList += "<td>" + clientObject.postalCode + "</td>";
          clientList += "<td>" + clientObject.phone + "</td>";
          clientList += "<td>" + clientObject.type + "</td>";
          clientList += "</tr>";
        }
      });
      $("tbody").html(clientList);
    });
  });
});
```
